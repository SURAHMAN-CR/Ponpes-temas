üìä Schema Database MySQL

-- Tabel Users (Siswa, Guru, Admin)
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nis VARCHAR(20) UNIQUE,
    nama VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role ENUM('siswa', 'guru', 'admin', 'ortu') DEFAULT 'siswa',
    jenis_kelamin ENUM('L', 'P'),
    tanggal_lahir DATE,
    tempat_lahir VARCHAR(100),
    alamat TEXT,
    no_telepon VARCHAR(20),
    foto_profil VARCHAR(255),
    kelas VARCHAR(10),
    nama_ayah VARCHAR(100),
    nama_ibu VARCHAR(100),
    pekerjaan_ayah VARCHAR(100),
    pekerjaan_ibu VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_role (role),
    INDEX idx_kelas (kelas)
);

-- Tabel Mata Pelajaran
CREATE TABLE mata_pelajaran (
    id INT PRIMARY KEY AUTO_INCREMENT,
    kode_mapel VARCHAR(10) UNIQUE NOT NULL,
    nama_mapel VARCHAR(100) NOT NULL,
    tingkat_kelas VARCHAR(20),
    deskripsi TEXT,
    guru_pengampu INT,
    FOREIGN KEY (guru_pengampu) REFERENCES users(id)
);

-- Tabel Nilai
CREATE TABLE nilai (
    id INT PRIMARY KEY AUTO_INCREMENT,
    siswa_id INT NOT NULL,
    mapel_id INT NOT NULL,
    jenis ENUM('tugas', 'quiz', 'uts', 'uas', 'praktikum'),
    nilai DECIMAL(5,2),
    tanggal DATE,
    semester VARCHAR(10),
    tahun_ajaran VARCHAR(9),
    catatan TEXT,
    FOREIGN KEY (siswa_id) REFERENCES users(id),
    FOREIGN KEY (mapel_id) REFERENCES mata_pelajaran(id),
    INDEX idx_siswa (siswa_id),
    INDEX idx_mapel (mapel_id)
);

-- Tabel Kehadiran
CREATE TABLE kehadiran (
    id INT PRIMARY KEY AUTO_INCREMENT,
    siswa_id INT NOT NULL,
    tanggal DATE NOT NULL,
    status ENUM('hadir', 'sakit', 'izin', 'alpha'),
    jam_masuk TIME,
    jam_puluh TIME,
    keterangan TEXT,
    FOREIGN KEY (siswa_id) REFERENCES users(id),
    UNIQUE KEY unique_attendance (siswa_id, tanggal),
    INDEX idx_tanggal (tanggal)
);

-- Tabel Materi Pembelajaran
CREATE TABLE materi (
    id INT PRIMARY KEY AUTO_INCREMENT,
    judul VARCHAR(200) NOT NULL,
    mapel_id INT,
    konten TEXT,
    tipe ENUM('pdf', 'video', 'ppt', 'link', 'text'),
    file_url VARCHAR(255),
    durasi INT, -- dalam menit
    tingkat_kesulitan ENUM('mudah', 'sedang', 'sulit'),
    poin INT DEFAULT 0,
    created_by INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (mapel_id) REFERENCES mata_pelajaran(id),
    FOREIGN KEY (created_by) REFERENCES users(id)
);

-- Tabel Tugas
CREATE TABLE tugas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    judul VARCHAR(200) NOT NULL,
    mapel_id INT,
    deskripsi TEXT,
    file_instruksi VARCHAR(255),
    deadline DATETIME,
    poin_max INT DEFAULT 100,
    created_by INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (mapel_id) REFERENCES mata_pelajaran(id),
    FOREIGN KEY (created_by) REFERENCES users(id)
);

-- Tabel Pengumpulan Tugas
CREATE TABLE pengumpulan_tugas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    tugas_id INT NOT NULL,
    siswa_id INT NOT NULL,
    file_url VARCHAR(255),
    jawaban_text TEXT,
    submitted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    nilai DECIMAL(5,2),
    dikoreksi_by INT,
    catatan TEXT,
    status ENUM('dikumpulkan', 'terlambat', 'belum', 'dikoreksi'),
    FOREIGN KEY (tugas_id) REFERENCES tugas(id),
    FOREIGN KEY (siswa_id) REFERENCES users(id),
    FOREIGN KEY (dikoreksi_by) REFERENCES users(id),
    UNIQUE KEY unique_submission (tugas_id, siswa_id)
);

-- Tabel Forum Diskusi
CREATE TABLE forum (
    id INT PRIMARY KEY AUTO_INCREMENT,
    topik VARCHAR(200) NOT NULL,
    konten TEXT,
    created_by INT,
    mapel_id INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (created_by) REFERENCES users(id),
    FOREIGN KEY (mapel_id) REFERENCES mata_pelajaran(id)
);

-- Tabel Komentar Forum
CREATE TABLE komentar_forum (
    id INT PRIMARY KEY AUTO_INCREMENT,
    forum_id INT NOT NULL,
    user_id INT NOT NULL,
    komentar TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (forum_id) REFERENCES forum(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- Tabel Pengumuman
CREATE TABLE pengumuman (
    id INT PRIMARY KEY AUTO_INCREMENT,
    judul VARCHAR(200) NOT NULL,
    konten TEXT NOT NULL,
    tipe ENUM('umum', 'akademik', 'kesiswaan', 'ortu'),
    target_kelas VARCHAR(100),
    created_by INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at DATE,
    FOREIGN KEY (created_by) REFERENCES users(id)
);

-- Tabel Event/Kegiatan
CREATE TABLE events (
    id INT PRIMARY KEY AUTO_INCREMENT,
    judul VARCHAR(200) NOT NULL,
    deskripsi TEXT,
    tanggal_mulai DATETIME,
    tanggal_selesai DATETIME,
    lokasi VARCHAR(200),
    tipe ENUM('akademik', 'ekstrakurikuler', 'keagamaan', 'lainnya'),
    peserta VARCHAR(100),
    created_by INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (created_by) REFERENCES users(id)
);

-- Tabel Pembayaran (SPP)
CREATE TABLE pembayaran (
    id INT PRIMARY KEY AUTO_INCREMENT,
    siswa_id INT NOT NULL,
    bulan VARCHAR(20),
    tahun VARCHAR(4),
    jumlah DECIMAL(10,2) NOT NULL,
    tanggal_bayar DATE,
    metode_bayar ENUM('transfer', 'tunai', 'qris'),
    status ENUM('lunas', 'belum', 'terlambat'),
    bukti_bayar VARCHAR(255),
    verified_by INT,
    catatan TEXT,
    FOREIGN KEY (siswa_id) REFERENCES users(id),
    FOREIGN KEY (verified_by) REFERENCES users(id),
    UNIQUE KEY unique_payment (siswa_id, bulan, tahun)
);
